

```bash
#!/bin/bash

# 5 * * * *  -- 每小时执行一次
# 5 0 * * *  -- 每天执行一次
LOG_PATH="/opt/test"
TODAY=`date +%Y%m%d`
#LASTDAY=`date -d '-7 day' "+%Y%m%d"`

# compress log
compress_log(){
for log in $(find $LOG_PATH -type f -name "*.log")
do
    gzip -c $log > $log-$TODAY.gz && > $log
done
}

# clear log
clear_log(){
    #find $LOG_PATH -type f -name "*-$LASTDAY.gz" | xargs rm -rf
    find $LOG_PATH -type f -name "*.gz" -mtime +7 | xargs rm -rf
}

compress_log
clear_log

```
