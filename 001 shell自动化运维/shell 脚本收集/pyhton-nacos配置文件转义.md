#!/bin/python
import sys
import os
import urllib
nacosFilePath='/tmp/nacos'
nacosFiles = os.listdir(nacosFilePath)
for fileName in nacosFiles:
    with open(nacosFilePath + '/' + fileName,'r+w') as f:
        text = f.read()
        urltext = urllib.quote(text.decode(sys.stdin.encoding).encode('utf8'))
        f.seek(0)
        f.write(urltext)
